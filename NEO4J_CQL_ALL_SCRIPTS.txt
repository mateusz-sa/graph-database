MATCH (c:Club)<-[PLAYS_FOR]-(Player)
RETURN c.name AS Club, count(*) AS Internationals
ORDER BY Internationals DESC LIMIT 10;

MATCH (c:Club)<-[PLAYS_FOR]-(p:Player)-[REPRESENTS]->(n:Country)
WHERE c.name='Manchester United'
RETURN n.name AS Country, count(*) AS Count, collect(p.name) AS Internationals
ORDER BY Count DESC;


Zapytanie zwraca Kraje wraz z sumą rozegranych meczych przez zawodników tego kraju których wiek jest większy niż 25 lat
MATCH (n:Country)<-[r:REPRESENTS]-(p:Player)
WHERE p.age >= 25
RETURN n.name AS Country, sum(r.numberOfGames) AS Games
ORDER BY Games DESC LIMIT 100;

Zapytanie zwraca piłkarzy (nazwa nie jest równa null) grających na pozycji Napastnik "Striker" wraz z krajem który reprezentują "Country"
MATCH (p:Player)
WHERE p.name IS NOT null
WITH p
MATCH (p)-[:PLAYS_ON]-(ps:Position)
WHERE ps.name IN ['Striker']
WITH p
MATCH (p)-[:REPRESENTS]-(n:Country)
RETURN DISTINCT p.name AS Player, n.name AS Country

Zapytanie znajdujące wwszystkie wezły od podanego o okresloną wartość
MATCH length=(p:Player)-[r]->(n)
RETURN length

Zapytanie znadujące odleglość pomiedzy wezłami Piłkarz i Druzyna narodowa (Kraj) w postaci liczby rozegranych meczy dla druzyny narodowej
MATCH (p:Player)-[r:REPRESENTS]->(n:Country)
RETURN p.name AS Player,   r.numberOfGames AS GamesTaken


WITH ['Striker','Winger'] AS playersOnPositionToBuy
UNWIND playersOnPositionToBuy AS position
MATCH (p:Player)-[r:PLAYS_ON]-(ps:Position {name: position})
WITH ps.name AS position, p.name AS player
ORDER BY position, player
RETURN position, collect(player) AS playersToBuy
